# useTooltipTrigger

<PageDescription>{docs.exports.useTooltipTrigger.description}</PageDescription>

`useTooltipTrigger` and `useTooltip` are the tooltip primitives, so the tooltip is built from scratch and positioned next to its trigger. The trigger receives the hook's props directly; here it is styled to match the [Button](../Button) component from React Aria Components.

```tsx render
"use client";
import React from 'react';
import {useTooltip, useTooltipTrigger} from 'react-aria/useTooltipTrigger';
import {useTooltipTriggerState} from 'react-stately/useTooltipTriggerState';
import {Overlay} from 'react-aria/Overlay';
import {mergeProps} from 'react-aria/mergeProps';
import 'vanilla-starter/Button.css';
import 'vanilla-starter/theme.css';

function Tooltip({state, ...props}: Parameters<typeof useTooltip>[0] & {state: ReturnType<typeof useTooltipTriggerState>, children?: React.ReactNode}) {
  /*- begin highlight -*/
  let {tooltipProps} = useTooltip(props, state);
  /*- end highlight -*/

  if (!state.isOpen) {
    return null;
  }

  return (
    <Overlay>
      <span
        {...mergeProps(props, tooltipProps)}
        style={{
          ...tooltipProps.style,
          background: 'var(--overlay-background)',
          color: 'var(--text-color)',
          border: '1px solid var(--gray-300)',
          borderRadius: 6,
          padding: '4px 8px',
          whiteSpace: 'nowrap',
          zIndex: 100,
          font: 'var(--font-size) system-ui',
          boxShadow: '0 8px 20px rgba(0, 0, 0, 0.1)'
        }}>
        {props.children}
      </span>
    </Overlay>
  );
}

function TooltipButton(props: Parameters<typeof useTooltipTrigger>[0] & {children?: React.ReactNode, tooltip?: React.ReactNode}) {
  let state = useTooltipTriggerState(props);
  let ref = React.useRef<HTMLButtonElement>(null);

  /*- begin highlight -*/
  let {triggerProps, tooltipProps} = useTooltipTrigger(props, state, ref);
  /*- end highlight -*/

  return (
    <>
      <button
        {...triggerProps}
        ref={ref}
        className="react-aria-Button button-base"
        data-variant="secondary">
        {props.children}
      </button>
      <Tooltip state={state} {...tooltipProps}>{props.tooltip}</Tooltip>
    </>
  );
}

<div style={{display: 'flex', gap: 8}}>
  <TooltipButton tooltip="Edit">Edit</TooltipButton>
  <TooltipButton tooltip="Delete">Delete</TooltipButton>
</div>
```
