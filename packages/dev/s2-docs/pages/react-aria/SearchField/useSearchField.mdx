# useSearchField

<PageDescription>{docs.exports.useSearchField.description}</PageDescription>

This example builds a search field from scratch with `useSearchField`, reusing the [Button](../Button) component from React Aria Components for the clear button, which appears once text is entered.

```tsx render
"use client";
import React from 'react';
import {useSearchField} from 'react-aria/useSearchField';
import {useFocus} from 'react-aria/useFocus';
import {mergeProps} from 'react-aria/mergeProps';
import {useSearchFieldState} from 'react-stately/useSearchFieldState';
import {Button} from 'react-aria-components/Button';
import {Label} from 'vanilla-starter/Form';
import {Search, X} from 'lucide-react';
import 'vanilla-starter/SearchField.css';

function SearchField(props) {
  // useSearchFieldState tracks the current text value.
  let state = useSearchFieldState(props);
  let ref = React.useRef(null);
  let {labelProps, inputProps, clearButtonProps} = useSearchField(props, state, ref);
  let [isFocused, setFocused] = React.useState(false);
  let {focusProps} = useFocus({onFocusChange: setFocused});

  return (
    <div className="react-aria-SearchField" data-empty={state.value === '' || undefined} style={{width: 250}}>
      <Label {...labelProps}>{props.label}</Label>
      <Search size={18} aria-hidden="true" />
      <input
        {...mergeProps(inputProps, focusProps)}
        ref={ref}
        className="react-aria-Input inset"
        data-focused={isFocused || undefined} />
      {/* The clear button shows when there is text (hidden via [data-empty] in the CSS). */}
      <Button {...clearButtonProps} className="clear-button">
        <X size={14} />
      </Button>
    </div>
  );
}

<SearchField label="Search" onSubmit={text => alert(text)} />
```
