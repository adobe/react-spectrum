## Value

Use the `value` or `defaultValue` prop to set the text value, and `onChange` to handle user input.

```tsx render
"use client";
import {TextArea} from '@react-spectrum/s2/TextArea';
import {useState} from 'react';

function Example() {
  let [name, setName] = useState('');

  return (
    <>
      <TextArea
        label="Comment"
        placeholder="Share your thoughts!"
        ///- begin highlight -///
        value={name}
        onChange={setName} />
      {/*- end highlight -*/}
      <p>Your comment: {name}</p>
    </>
  );
}
```

## Prefix

Use the `prefix` prop to display a prefix in the text area. Either a string or workflow icon.

```tsx render
"use client";
import {TextArea} from '@react-spectrum/s2/TextArea';
import MentionIcon from '@react-spectrum/s2/icons/Mention';
import {style} from '@react-spectrum/s2/style' with {type: 'macro'};

function Example() {
  return (
    <div className={style({display: 'flex', flexDirection: 'column', gap: 8})}>
      <TextArea label="Irrational number" prefix="#" placeholder="3.1415926535897932384626433832795028841971" />
      <TextArea label="Mention" prefix={<MentionIcon />} placeholder="username" />
    </div>
  );
}
```

## Forms

Use the `name` prop to submit the text value to the server. Set the `isRequired`, `minLength`, or `maxLength` props to validate the value, or implement custom client or server-side validation. See the [Forms](forms) guide to learn more.

```tsx render docs={docs.exports.TextArea} links={docs.links} props={['isRequired', 'necessityIndicator', 'minLength', 'maxLength']} initialProps={{isRequired: true, minLength: 10}}
"use client";
import {TextArea} from '@react-spectrum/s2/TextArea';
import {Button} from '@react-spectrum/s2/Button';
import {Form} from '@react-spectrum/s2/Form';

function Example(props) {
  return (
    <Form>
      <TextArea
        {...props}
        label="Comment"
        placeholder="Share your thoughts!"
        ///- begin highlight -///
        name="email"
        description="Please provide at least 10 characters."
        /* PROPS */
        ///- end highlight -///
      />
      <Button type="submit">Submit</Button>
    </Form>
  );
}
```
